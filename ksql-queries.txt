CREATE STREAM POD_TELEMETRY_102 AS \ SELECT VEL, ACCEL FROM POD_TELEMETRY \
WHERE POD_ID=102;

CREATE STREAM POD_TELEMETRY_RAW (POD_ID int, VEL double, ACCEL double) \
WITH (KAFKA_TOPIC='pod_telemetry_raw', VALUE_FORMAT='JSON');

CREATE STREAM POD_TELEMETRY \
WITH (VAlUE_FORMAT='AVRO', KAFKA_TOPIC='pod_telemetry_avro') \
AS SELECT * FROM POD_TELEMETRY_RAW;
